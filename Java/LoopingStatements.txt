Question 3: Fibonacci
import java.util.Scanner;

public class Main {

	public static void main(String[] args) {

		Scanner input = new Scanner(System.in);
		int n = input.nextInt();

		int a = -1;
		int b = 1;
		int temp = 0;
		int flag = 0;
		for (int i = 0; temp <= n && flag == 0; i++) {

			temp = a + b;
			a = b;
			b = temp;
			if (n == temp) {
				flag = 1;
			} else {
				flag = 0;
			}
		}
		String str = (flag == 1) ? "yes" : "no";
		System.out.println(str);
	}

}
